CREATE TABLE MEMBER(
    ID VARCHAR2(20) PRIMARY KEY,
    PWD VARCHAR2(20),
    NAME VARCHAR2(5),
    EMAIL VARCHAR2(10)
);

CREATE TABLE SITE(
    ID VARCHAR2(20),
    VTIME DATE,
    SITE VARCHAR2(1000),
    SEARCH VARCHAR2(1000),
    FOREIGN KEY (ID) REFERENCES MEMBER(ID)
);
